59time_duration estimate_effect_dur( int skill_lvl, const efftype_id &target_effect,
60 const time_duration &error_magnitude, int threshold, const Creature &target )
61{
62 const time_duration zero_duration = 0_turns;

64 time_duration estimate = std::max( zero_duration, target.get_effect_dur( target_effect ) +
65 rng( -1, 1 ) * error_magnitude *
66 rng( 0, std::max( 0, threshold - skill_lvl ) ) );
67 return estimate;

138float fine_detail_vision_mod( const Character &who, const tripoint &p )
139{
140 // PER_SLIME_OK implies you can get enough eyes around the bile
141 // that you can generally see. There still will be the haze, but
142 // it's annoying rather than limiting.
143 if( who.is_blind() ||
144 ( ( who.has_effect( effect_boomered ) || who.has_effect( effect_darkness ) ) &&
145 !who.has_trait( trait_PER_SLIME_OK ) ) ) {
146 return 11.0;
147 }
148 // Scale linearly as light level approaches LIGHT_AMBIENT_LIT.
149 // If we're actually a source of light, assume we can direct it where we need it.
150 // Therefore give a hefty bonus relative to ambient light.
151 float own_light = std::max( 1.0f, LIGHT_AMBIENT_LIT - who.active_light() - 2.0f );
152
153 // Same calculation as above, but with a result 3 lower.
154 float ambient_light = std::max( 1.0f, LIGHT_AMBIENT_LIT - get_map().ambient_light_at( p ) + 1.0f );
155
156 return std::min( own_light, ambient_light );
157}

169comfort_response_t base_comfort_value( const Character &who, const tripoint &p )
170{
171 // Comfort of sleeping spots is "objective", while sleep_spot( p ) is "subjective"
172 // As in the latter also checks for fatigue and other variables while this function
173 // only looks at the base comfyness of something. It's still subjective, in a sense,
174 // as arachnids who sleep in webs will find most places comfortable for instance.
175 int comfort = 0;
176
177 comfort_response_t comfort_response;
178
179 bool plantsleep = who.has_trait( trait_CHLOROMORPH );
180 bool fungaloid_cosplay = who.has_trait( trait_M_SKIN3 );
181 bool websleep = who.has_trait( trait_WEB_WALKER );
182 bool webforce = who.has_trait( trait_THRESH_SPIDER ) && ( who.has_trait( trait_WEB_SPINNER ) ||
183 ( who.has_trait( trait_WEB_WEAVER ) ) );
184 bool in_shell = who.has_active_mutation( trait_SHELL2 );
185 bool watersleep = who.has_trait( trait_WATERSLEEP );
186
187 map &here = get_map();
188 const optional_vpart_position vp = here.veh_at( p );
189 const maptile tile = here.maptile_at( p );
190 const trap &trap_at_pos = tile.get_trap_t();
191 const ter_id ter_at_pos = tile.get_ter();
192 const furn_id furn_at_pos = tile.get_furn();
193
194 int web = here.get_field_intensity( p, fd_web );
195
196 // Some mutants have different comfort needs
197 if( !plantsleep && !webforce ) {
198 if( in_shell ) {
199 comfort += 1 + static_cast<int>( comfort_level::slightly_comfortable );
200 // Note: shelled individuals can still use sleeping aids!
201 } else if( vp ) {
202 const cata::optional<vpart_reference> carg = vp.part_with_feature( "CARGO", false );
203 const cata::optional<vpart_reference> board = vp.part_with_feature( "BOARDABLE", true );
204 if( carg ) {
205 const vehicle_stack items = vp->vehicle().get_items( carg->part_index() );
206 for( const item &items_it : items ) {
207 if( items_it.has_flag( "SLEEP_AID" ) ) {
208 // Note: BED + SLEEP_AID = 9 pts, or 1 pt below very_comfortable
209 comfort += 1 + static_cast<int>( comfort_level::slightly_comfortable );
210 comfort_response.aid = &items_it;
211 break; // prevents using more than 1 sleep aid
212 }
213 }
214 }
215 if( board ) {
216 comfort += board->info().comfort;
217 } else {
218 comfort -= here.move_cost( p );
219 }
220 }
221 // Not in a vehicle, start checking furniture/terrain/traps at this point in decreasing order
222 else if( furn_at_pos != f_null ) {
223 comfort += 0 + furn_at_pos.obj().comfort;
224 }
225 // Web sleepers can use their webs if better furniture isn't available
226 else if( websleep && web >= 3 ) {
227 comfort += 1 + static_cast<int>( comfort_level::slightly_comfortable );
228 } else if( ter_at_pos == t_improvised_shelter ) {
229 comfort += 0 + static_cast<int>( comfort_level::slightly_comfortable );
230 } else if( ter_at_pos == t_floor || ter_at_pos == t_floor_waxed ||
231 ter_at_pos == t_carpet_red || ter_at_pos == t_carpet_yellow ||
232 ter_at_pos == t_carpet_green || ter_at_pos == t_carpet_purple ) {
233 comfort += 1 + static_cast<int>( comfort_level::neutral );
234 } else if( !trap_at_pos.is_null() ) {
235 comfort += 0 + trap_at_pos.comfort;
236 } else {
237 // Not a comfortable sleeping spot
238 comfort -= here.move_cost( p );
239 }
240
241 if( comfort_response.aid == nullptr ) {
242 const map_stack items = here.i_at( p );
243 for( const item &items_it : items ) {
244 if( items_it.has_flag( "SLEEP_AID" ) ) {
245 // Note: BED + SLEEP_AID = 9 pts, or 1 pt below very_comfortable
246 comfort += 1 + static_cast<int>( comfort_level::slightly_comfortable );
247 comfort_response.aid = &items_it;
248 break; // prevents using more than 1 sleep aid
249 }
250 }
251 }
252 if( fungaloid_cosplay && here.has_flag_ter_or_furn( flag_FUNGUS, p ) ) {
253 comfort += static_cast<int>( comfort_level::very_comfortable );
254 } else if( watersleep && here.has_flag_ter( flag_SWIMMABLE, p ) ) {
255 comfort += static_cast<int>( comfort_level::very_comfortable );
256 }
257 } else if( plantsleep ) {
258 if( vp || furn_at_pos != f_null ) {
259 // Sleep ain't happening in a vehicle or on furniture
260 comfort = static_cast<int>( comfort_level::impossible );
261 } else {
262 // It's very easy for Chloromorphs to get to sleep on soil!
263 if( ter_at_pos == t_dirt || ter_at_pos == t_pit || ter_at_pos == t_dirtmound ||
264 ter_at_pos == t_pit_shallow ) {
265 comfort += static_cast<int>( comfort_level::very_comfortable );
266 }
267 // Not as much if you have to dig through stuff first
268 else if( ter_at_pos == t_grass ) {
269 comfort += static_cast<int>( comfort_level::comfortable );
270 }
271 // Sleep ain't happening
272 else {
273 comfort = static_cast<int>( comfort_level::impossible );
274 }
275 }
276 // Has webforce
277 } else {
278 if( web >= 3 ) {
279 // Thick Web and you're good to go
280 comfort += static_cast<int>( comfort_level::very_comfortable );
281 } else {
282 comfort = static_cast<int>( comfort_level::impossible );
283 }
284 }
285
286 if( comfort > static_cast<int>( comfort_level::comfortable ) ) {
287 comfort_response.level = comfort_level::very_comfortable;
288 } else if( comfort > static_cast<int>( comfort_level::slightly_comfortable ) ) {
289 comfort_response.level = comfort_level::comfortable;
290 } else if( comfort > static_cast<int>( comfort_level::neutral ) ) {
291 comfort_response.level = comfort_level::slightly_comfortable;
292 } else if( comfort == static_cast<int>( comfort_level::neutral ) ) {
293 comfort_response.level = comfort_level::neutral;
294 } else {
295 comfort_response.level = comfort_level::uncomfortable;
296 }
297 return comfort_response;
298}

300int rate_sleep_spot( const Character &who, const tripoint &p )
301{
302 const int current_stim = who.get_stim();
303 const comfort_response_t comfort_info = base_comfort_value( who, p );
304 if( comfort_info.aid != nullptr ) {
305 who.add_msg_if_player( m_info, _( "You use your %s for comfort." ), comfort_info.aid->tname() );
306 }
307
308 int sleepy = static_cast<int>( comfort_info.level );
309 bool watersleep = who.has_trait( trait_WATERSLEEP );
310
311 if( who.has_addiction( add_type::SLEEP ) ) {
312 sleepy -= 4;
313 }
314 if( who.has_trait( trait_INSOMNIA ) ) {
315 // 12.5 points is the difference between "tired" and "dead tired"
316 sleepy -= 12;
317 }
318 if( who.has_trait( trait_EASYSLEEPER ) ) {
319 // Low fatigue (being rested) has a much stronger effect than high fatigue
320 // so it's OK for the value to be that much higher
321 sleepy += 40;
322 }
323 if( who.has_active_bionic( bio_soporific ) ) {
324 sleepy += 30;
325 }
326 if( who.has_trait( trait_EASYSLEEPER2 ) ) {
327 // At this point, the only limit to sleep is tiredness
328 sleepy += 100;
329 }
330 if( watersleep && get_map().has_flag_ter( "SWIMMABLE", p ) ) {
331 sleepy += 10; //comfy water!
332 }
333
334 if( who.get_fatigue() < fatigue_levels::tired + 1 ) {
335 sleepy -= static_cast<int>( ( fatigue_levels::tired + 1 - who.get_fatigue() ) / 4 );
336 } else {
337 sleepy += static_cast<int>( ( who.get_fatigue() - fatigue_levels::tired + 1 ) / 16 );
338 }
339
340 if( current_stim > 0 || !who.has_trait( trait_INSOMNIA ) ) {
341 sleepy -= 2 * current_stim;
342 } else {
343 // Make it harder for insomniac to get around the trait
344 sleepy -= current_stim;
345 }
346
347 return sleepy;
348}

350bool roll_can_sleep( Character &who )
351{
352 if( who.has_effect( effect_meth ) ) {
353 // Sleep ain't happening until that meth wears off completely.
354 return false;
355 }
356
357 // Since there's a bit of randomness to falling asleep, we want to
358 // prevent exploiting this if can_sleep() gets called over and over.
359 // Only actually check if we can fall asleep no more frequently than
360 // every 30 minutes. We're assuming that if we return true, we'll
361 // immediately be falling asleep after that.
362 //
363 // Also if player debug menu'd time backwards this breaks, just do the
364 // check anyway, this will reset the timer if 'dur' is negative.
365 const time_point now = calendar::turn;
366 const time_duration dur = now - who.last_sleep_check;
367 if( dur >= 0_turns && dur < 30_minutes ) {
368 return false;
369 }
370 who.last_sleep_check = now;
371
372 int sleepy = character_funcs::rate_sleep_spot( who, who.pos() );
373 sleepy += rng( -8, 8 );
374 bool result = sleepy > 0;
375
376 if( who.has_active_bionic( bio_soporific ) ) {
377 if( who.bio_soporific_powered_at_last_sleep_check && !who.has_power() ) {
378 who.add_msg_if_player( m_bad, _( "Your soporific inducer runs out of power!" ) );
379 } else if( !who.bio_soporific_powered_at_last_sleep_check && who.has_power() ) {
380 who.add_msg_if_player( m_good, _( "Your soporific inducer starts back up." ) );
381 }
382 who.bio_soporific_powered_at_last_sleep_check = who.has_power();
383 }
384
385 return result;
386}

590cata::optional<tripoint> pick_safe_adjacent_tile( const Character &who )
591{
592 std::vector<tripoint> ret;
593 int dangerous_fields = 0;
594 map &here = get_map();
595 for( const tripoint &p : here.points_in_radius( who.pos(), 1 ) ) {
596 if( p == who.pos() ) {
597 // Don't consider player position
598 continue;
599 }
600 const trap &curtrap = here.tr_at( p );
601 if( g->critter_at( p ) == nullptr && here.passable( p ) &&
602 ( curtrap.is_null() || curtrap.is_benign() ) ) {
603 // Only consider tile if unoccupied, passable and has no traps
604 dangerous_fields = 0;
605 auto &tmpfld = here.field_at( p );
606 for( auto &fld : tmpfld ) {
607 const field_entry &cur = fld.second;
608 if( cur.is_dangerous() ) {
609 dangerous_fields++;
610 }
611 }
612
613 if( dangerous_fields == 0 && ! get_map().obstructed_by_vehicle_rotation( who.pos(), p ) ) {
614 ret.push_back( p );
615 }
616 }
617 }
618
619 return random_entry( ret );
620}

622bool is_bp_immune_to( const Character &who, body_part bp, damage_unit dam )
623{
624 if( who.has_trait( trait_DEBUG_NODMG ) || who.is_immune_damage( dam.type ) ) {
625 return true;
626 }
627
628 who.passive_absorb_hit( convert_bp( bp ).id(), dam );
629
630 for( const item &cloth : who.worn ) {

632 cloth.mitigate_damage( dam );
633 }
634 }
635
636 return dam.amount <= 0;
